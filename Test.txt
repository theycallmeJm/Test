import React, { useState, useEffect } from 'react';
import { useHistory } from 'react-router-dom';

const EditTaskForm = ({ task, onUpdateTask }) => {
  const history = useHistory();
  const [formData, setFormData] = useState({
    title: task.title,
    description: task.description,
    status: task.status,
    dueDate: task.dueDate, // Assuming task.dueDate is a date string
  });

  const handleInputChange = (e) => {
    const { name, value } = e.target;
    setFormData({
      ...formData,
      [name]: value,
    });
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    onUpdateTask({
      ...task,
      ...formData,
    });
    history.push('/');
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label>Title:</label>
        <input
          type="text"
          name="title"
          value={formData.title}
          onChange={handleInputChange}
        />
      </div>
      <div>
        <label>Description:</label>
        <textarea
          name="description"
          value={formData.description}
          onChange={handleInputChange}
        ></textarea>
      </div>
      <div>
        <label>Status:</label>
        <select
          name="status"
          value={formData.status ? 'Complete' : 'Incomplete'}
          onChange={(e) => {
            const newValue = e.target.value === 'Complete' ? true : false;
            handleInputChange({ target: { name: 'status', value: newValue } });
          }}
        >
          <option value="Incomplete">Incomplete</option>
          <option value="Complete">Complete</option>
        </select>
      </div>
      <div>
        <label>Due Date:</label>
        <input
          type="date"
          name="dueDate"
          value={formData.dueDate}
          onChange={handleInputChange}
        />
      </div>
      <button type="submit">Update Task</button>
    </form>
  );
};

export default EditTaskForm;
