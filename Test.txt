<div>
  <label>Status:</label>
  <select
    name="status"
    value={formData.status ? 'Complete' : 'Incomplete'} // Set value based on formData.status
    onChange={(e) => {
      const newValue = e.target.value === 'Complete' ? true : false; // Convert value to boolean
      handleInputChange({ target: { name: 'status', value: newValue } }); // Call handleInputChange with updated value
    }}
  >
    <option value="Incomplete">Incomplete</option>
    <option value="Complete">Complete</option>
  </select>
</div>
