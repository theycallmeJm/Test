import React, { useState } from 'react';
import TaskList from './TaskList';
import TaskDetail from './TaskDetail';
import AddTaskForm from './AddTaskForm';

const App = () => {
  const [tasks, setTasks] = useState([]); // Assuming tasks state is managed here

  // Function to handle form submission and add a new task
  const handleAddTask = (newTask) => {
    setTasks([...tasks, newTask]);
  };

  // Assuming other functions and state for task management are defined here

  return (
    <div>
      <h1>Task Manager</h1>
      <AddTaskForm onSubmit={handleAddTask} latestId={tasks.length ? tasks[tasks.length - 1].id : 0} />
      <TaskList tasks={tasks} />
      <TaskDetail /> {/* Assuming TaskDetail component is rendered somewhere */}
    </div>
  );
};

export default App;
