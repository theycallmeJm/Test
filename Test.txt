import React, { useState, useEffect } from 'react';
import { BrowserRouter as Router, Route, Switch } from 'react-router-dom';
import axios from 'axios';
import TaskList from './TaskList';
import TaskDetail from './TaskDetail';
import EditTaskForm from './EditTaskForm'; // Import EditTaskForm component

const App = () => {
  const [tasks, setTasks] = useState([]);

  // Fetch tasks from mock API when component mounts
  useEffect(() => {
    const fetchTasks = async () => {
      try {
        const response = await axios.get('/tasks'); // Assuming /tasks is the endpoint for fetching tasks
        setTasks(response.data);
      } catch (error) {
        console.error('Error fetching tasks:', error);
      }
    };
    fetchTasks();
  }, []);

  // Function to update task in state
  const updateTaskInState = (updatedTask) => {
    const updatedTasks = tasks.map((t) =>
      t.id === updatedTask.id ? updatedTask : t
    );
    setTasks(updatedTasks);
  };

  return (
    <Router>
      <div>
        <h1>Task Manager</h1>
        <Switch>
          <Route exact path="/">
            <TaskList tasks={tasks} />
          </Route>
          <Route exact path="/tasks/:taskId">
            <TaskDetail tasks={tasks} />
          </Route>
          <Route exact path="/tasks/:taskId/edit">
            <EditTaskForm onUpdateTask={updateTaskInState} />
          </Route>
        </Switch>
      </div>
    </Router>
  );
};

export default App;
