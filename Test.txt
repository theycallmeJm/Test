// TaskList.jsx

import React from 'react';
import { Link } from 'react-router-dom';
import './TaskList.css'; // Import the CSS file for TaskList

function TaskList({ tasks }) {
  return (
    <div className="task-list-container">
      <h1>To-Do List</h1>
      <div className="task-grid">
        {tasks.map((task) => (
          <div className={`task-card ${task.completed ? 'completed' : 'incomplete'}`} key={task.id}>
            <div className="task-header">
              <h3 className="task-title">{task.title}</h3>
              <span className="task-status">{task.completed ? 'Completed' : 'Incomplete'}</span>
            </div>
            <div className="task-footer">
              <Link className="task-button" to={`/tasks/${task.id}`}>View</Link>
              <button className="task-button">Edit</button>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
}

export default TaskList;
